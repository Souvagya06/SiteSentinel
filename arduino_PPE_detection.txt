#include <WiFi.h>
#include <WebServer.h>

const char* ssid     = "Souvagya's HONOR 200";
const char* password = "Souv@gya2006";

WebServer server(80);

const int buzzerPin   = 23;
const int greenLedPin = 18;
const int redLedPin   = 19;

unsigned long greenOffAt = 0;
unsigned long redOffAt   = 0;

void alarmOn()  { digitalWrite(buzzerPin, HIGH); server.send(200, "text/plain", "Alarm ON"); }
void alarmOff() { digitalWrite(buzzerPin, LOW);  server.send(200, "text/plain", "Alarm OFF"); }

void checkinLight() {
    digitalWrite(greenLedPin, HIGH);
    digitalWrite(redLedPin,   LOW);
    greenOffAt = millis() + 3000;   // turn off after 3s non-blocking
    server.send(200, "text/plain", "Check-in light");
}

void checkoutLight() {
    digitalWrite(redLedPin,   HIGH);
    digitalWrite(greenLedPin, LOW);
    redOffAt = millis() + 3000;
    server.send(200, "text/plain", "Check-out light");
}

void setup() {
    Serial.begin(115200);
    pinMode(buzzerPin,   OUTPUT);
    pinMode(greenLedPin, OUTPUT);
    pinMode(redLedPin,   OUTPUT);
    digitalWrite(buzzerPin,   LOW);
    digitalWrite(greenLedPin, LOW);
    digitalWrite(redLedPin,   LOW);

    WiFi.begin(ssid, password);
    while (WiFi.status() != WL_CONNECTED) { delay(500); Serial.print("."); }
    Serial.println("\nConnected!");
    Serial.println(WiFi.localIP());

    server.on("/on",       alarmOn);
    server.on("/off",      alarmOff);
    server.on("/checkin",  checkinLight);
    server.on("/checkout", checkoutLight);
    server.begin();
}

void loop() {
    server.handleClient();

    // Non-blocking LED off
    if (greenOffAt && millis() > greenOffAt) {
        digitalWrite(greenLedPin, LOW);
        greenOffAt = 0;
    }
    if (redOffAt && millis() > redOffAt) {
        digitalWrite(redLedPin, LOW);
        redOffAt = 0;
    }
}
